<script>
(async () => {
  const params = new URLSearchParams(location.search);
  const ref = params.get("ref");
  const box = document.getElementById("payment-box");
  const title = document.getElementById("payment-title");
  const copy = document.getElementById("payment-copy");

  if (!ref) {
    title.textContent = "Payment reference missing";
    box.textContent = "Please return to the homepage and start the booking through Ken.";
    return;
  }

  async function check() {
    const response = await fetch("/api/payment-status?ref=" + encodeURIComponent(ref));
    const data = await response.json();
    if (!response.ok) throw new Error(data.error || "Could not verify payment.");
    return data;
  }

  try {
    let data = await check();
    if (!data.paid) {
      await new Promise(r => setTimeout(r, 2500));
      data = await check();
    }

    if (!data.paid) {
      title.textContent = "Payment not confirmed yet";
      copy.textContent = "SumUp has not returned a paid status yet.";
      box.innerHTML = '<p>Your payment may still be processing. Refresh this page shortly. If you were charged and this does not update, call 020 7371 3333.</p>';
      return;
    }

    title.textContent = "Payment confirmed — choose your preferred visit";
    copy.textContent = "Your £75 attendance and diagnosis payment is confirmed.";
    box.innerHTML = `
      <form id="booking-form" class="form-grid">
        <div class="field"><label>Preferred date</label><input name="preferredDate" type="date" required></div>
        <div class="field"><label>Preferred time</label><select name="preferredWindow" required><option value="">Choose…</option><option>8am–12pm</option><option>12pm–4pm</option><option>4pm–7pm</option></select></div>
        <div class="field full"><label>Anything else Ken should pass to the plumber?</label><textarea name="notes" placeholder="Access instructions, parking, floor/flat number, anything that may help."></textarea></div>
        <div class="field full"><button class="btn btn-primary" type="submit">Request this appointment</button></div>
      </form>`;

    document.getElementById("booking-form").addEventListener("submit", async event => {
      event.preventDefault();
      const form = event.currentTarget;
      const values = Object.fromEntries(new FormData(form).entries());
      const response = await fetch("/api/book", {
        method:"POST",
        headers:{"Content-Type":"application/json"},
        body:JSON.stringify({paymentId:data.paymentId,...values})
      });
      const result = await response.json();
      if (!response.ok) {
        alert(result.error || "Could not save booking.");
        return;
      }
      title.textContent = "Booking request received";
      copy.textContent = "Thank you. Kensington Plumbing Services has your preferred appointment details.";
      box.innerHTML = "<p><strong>Booking reference:</strong> " + result.bookingId + "</p><p>We will confirm the appointment details using the contact information you supplied to Ken.</p><p><a class='btn btn-dark' href='/'>Back to Kensington Plumbing Services</a></p>";
    });
  } catch(error) {
    title.textContent = "We could not verify the payment";
    box.textContent = error.message + " Please call 020 7371 3333 if you need help.";
  }
})();
</script>
